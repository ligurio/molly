sqlite-rw-register.lua

local sqlite3 = require('lsqlite3')
local ljepsen = require('ljepsen')
local math = require('math')
local os = require('os')

local client = ljepsen.client
local gen_lib = ljepsen.gen
local runner = ljepsen.runner

-- Function that implements a read opeation, it returns a name of operation and
-- a value equal to nil because we don't know it before read.
local function r()
    return {
        f = 'txn',
        value = {{
            'r',
            'x',
            nil,
        }},
    }
end

-- Function that implements a write opeation, it returns a name of operation
-- and random generated number.
local function w()
    return {
        f = 'txn',
        value = {{
            'w',
            'x',
            math.random(1, 100),
        }}
    }
end

local db = sqlite3.open_memory() -- Open SQLite database in memory.
db:exec[[ CREATE TABLE test (id, content) ]]

local sqlite_rw_register = client.new()

-- sqlite_rw_register implements a client that performs two operations on
-- database: read and write. Method invoke must apply these operations to a
-- database instance.
sqlite_rw_register.invoke = function(self, op)
    local val = op.value[1]
    if val[1] == 'r' then
        val[3] = db:exec([[ SELECT content FROM test WHERE id = 1 ]])
    elseif val[1] == 'w' then
        db:exec(string.format('INSERT INTO test VALUES (1, %d)', val[3]))
    else
        error('Unknown operation')
    end

    return {
        value = { val },
        f = op.f,
        process = op.process,
    }
end

local test_options = {
    -- Keep reports with history in files history.{json,txt}.
    create_reports = true,
    threads = 1,
}
local ok = runner.run_test({
    client = sqlite_rw_register,
    generator = gen_lib.cycle(gen_lib.iter({ r, w })):take(100)
}, test_options)

if ok == true then
    print('Feel free to check history.json with elle-cli:')
    print('java -jar elle-cli-0.1.1-standalone.jar --model elle-rw-register history.json')
end

local changes = db:total_changes()
assert(changes == 50, string.format('Number of operations is wrong (%d vs 50)', changes))
print('Total changes in SQLite DB:', changes)
db:close() -- Close a database handler.

if os.getenv('DEV') ~= 'ON' then
    os.remove('history.json')
    os.remove('history.txt')
end
